    <script>
        let currentStep = 1;
        const totalSteps = document.querySelectorAll('.form-step').length; 
        const form = document.getElementById('pilot-application-form');
        const nextButton = document.getElementById('next-button');
        const backButton = document.getElementById('back-button');
        const submitButton = document.getElementById('submit-button');
        const indicatorsContainer = document.getElementById('step-indicators-container');

        // --- UI Initialization and Updates ---

        function setupIndicators() {
            indicatorsContainer.innerHTML = '';
            for (let i = 1; i <= totalSteps; i++) {
                const indicator = document.createElement('div');
                indicator.id = `step-${i}-indicator`;
                indicator.className = 'step-indicator flex items-center justify-center rounded-full border-2 text-gray-500 font-bold transition-all duration-300';
                indicator.textContent = i;
                
                if (i > 1) {
                    const separator = document.createElement('div');
                    separator.className = 'flex-grow h-0.5 bg-gray-200 mx-1';
                    indicatorsContainer.appendChild(separator);
                }

                indicatorsContainer.appendChild(indicator);
            }
        }

        function updateUI() {
            document.querySelectorAll('.form-step').forEach(step => step.classList.remove('active'));
            document.getElementById(`step-${currentStep}`).classList.add('active');

            backButton.classList.toggle('hidden', currentStep === 1);
            nextButton.classList.toggle('hidden', currentStep === totalSteps);
            submitButton.classList.toggle('hidden', currentStep !== totalSteps);
            
            nextButton.classList.toggle('ml-auto', currentStep === 1);
            backButton.classList.toggle('ml-auto', currentStep !== 1);

            document.querySelectorAll('.step-indicator').forEach((indicator, index) => {
                indicator.classList.remove('active', 'complete', 'border-emerald-500', 'border-gray-300');
                const step = index + 1;
                if (step === currentStep) {
                    indicator.classList.add('active');
                } else if (step < currentStep) {
                    indicator.classList.add('complete');
                } else {
                    indicator.classList.add('border-2', 'border-gray-300', 'text-gray-500');
                }
            });
        }

        // --- Conditional Logic (Steps 5 to 11 Requirement) ---
        
        function toggleLicenseRequirement(isRequired) {
            const licenseInput = document.getElementById('attach_license');
            const requiredText = document.getElementById('license-required-text');

            if (isRequired) {
                licenseInput.setAttribute('required', 'required');
                requiredText.classList.remove('hidden');
            } else {
                licenseInput.removeAttribute('required');
                requiredText.classList.add('hidden');
                // Optional: Clear the file input if the requirement is removed
                licenseInput.value = ''; 
            }
        }

        // --- Conditional Input Visibility (Steps 6 & 7) ---

        function toggleExperienceInput(fieldName) {
            const radioGroupName = fieldName + '_radio';
            const inputContainer = document.getElementById(fieldName + '_input');
            const relatedInput = document.getElementById(fieldName);
            
            const selectedRadio = document.querySelector(`input[name="${radioGroupName}"]:checked`);
            
            if (selectedRadio && selectedRadio.value === 'Yes') {
                inputContainer.classList.remove('max-h-0', 'opacity-0');
                inputContainer.classList.add('max-h-96', 'opacity-100');
                if (relatedInput) relatedInput.setAttribute('required', 'required');
            } else {
                inputContainer.classList.remove('max-h-96', 'opacity-100');
                inputContainer.classList.add('max-h-0', 'opacity-0');
                if (relatedInput) {
                    relatedInput.removeAttribute('required');
                    relatedInput.value = '';
                }
            }
        }


        // --- Validation Logic (Client-side checks) ---

        function isRadioGroupChecked(name) {
            return document.querySelector(`input[name="${name}"]:checked`) !== null;
        }

        function validateStep(step) {
            const stepElement = document.getElementById(`step-${step}`);
            let isValid = true;
            
            const fields = stepElement.querySelectorAll('[required]');
            
            // Check required fields (including conditional file input in step 11)
            fields.forEach(field => {
                if (field.type !== 'radio' && field.type !== 'file' && !field.value.trim()) {
                    isValid = false;
                    field.reportValidity();
                } else if (field.type === 'file' && field.hasAttribute('required') && field.files.length === 0) {
                     isValid = false;
                     field.reportValidity();
                }
            });

            if (!isValid) return false;
            
            // --- Custom Validation Checks ---
            if (step === 1) {
                const fullNameField = document.getElementById('full_name');
                const nameError = document.getElementById('name-error');
                const namePattern = /^[a-zA-Z\s]+$/;
                if (!namePattern.test(fullNameField.value.trim())) {
                    nameError.classList.remove('hidden');
                    isValid = false;
                } else {
                    nameError.classList.add('hidden');
                }
            } 
            // Step 2 (Email) only checks basic browser validity here; uniqueness is server-side
            else if (step === 3) {
                const phoneField = document.getElementById('phone_number');
                const phoneError = document.getElementById('phone-error');
                const phonePattern = /^\d+$/; 
                if (!phonePattern.test(phoneField.value.trim())) {
                    phoneError.classList.remove('hidden');
                    isValid = false;
                } else {
                    phoneError.classList.add('hidden');
                }
                const countryCodeField = document.getElementById('country_code');
                const codePattern = /^\+\d{1,4}$/;
                if (!codePattern.test(countryCodeField.value.trim())) {
                    countryCodeField.setCustomValidity("Please enter a valid country code (e.g., +44).");
                    isValid = false;
                } else {
                     countryCodeField.setCustomValidity("");
                }
            } else if ([4, 5, 8, 9].includes(step)) {
                // Radio group required check
                const radioGroupName = stepElement.querySelector('input[type="radio"]').name;
                 if (!isRadioGroupChecked(radioGroupName)) {
                    stepElement.querySelector('input[type="radio"]').reportValidity();
                    isValid = false;
                }
            } else if (step === 6) {
                // Conditional text field validation
                if (!isRadioGroupChecked('prior_experience_radio')) {
                    document.getElementsByName('prior_experience_radio')[0].reportValidity();
                    isValid = false;
                } else if (document.getElementById('exp_yes').checked && !document.getElementById('prior_experience').value.trim()) {
                     document.getElementById('prior_experience').reportValidity();
                     isValid = false;
                }
            } else if (step === 7) {
                 // Conditional text field validation
                 if (!isRadioGroupChecked('own_drone_radio')) {
                    document.getElementsByName('own_drone_radio')[0].reportValidity();
                    isValid = false;
                } else if (document.getElementById('own_yes').checked && !document.getElementById('own_drone').value.trim()) {
                     document.getElementById('own_drone').reportValidity();
                     isValid = false;
                }
            }

            return isValid;
        }

        function nextStep() {
            if (validateStep(currentStep)) {
                if (currentStep < totalSteps) {
                    currentStep++;
                    updateUI();
                }
            }
        }

        function prevStep() {
            if (currentStep > 1) {
                currentStep--;
                updateUI();
            }
        }

        function resetForm() {
            form.reset();
            currentStep = 1;
            updateUI();
            document.getElementById('submission-message').classList.add('hidden');
            document.getElementById('submission-error').classList.add('hidden');
            document.getElementById('name-error').classList.add('hidden');
            document.getElementById('phone-error').classList.add('hidden');
            
            document.getElementById('prior_experience_input').classList.remove('max-h-96', 'opacity-100');
            document.getElementById('prior_experience_input').classList.add('max-h-0', 'opacity-0');
            document.getElementById('own_drone_input').classList.remove('max-h-96', 'opacity-100');
            document.getElementById('own_drone_input').classList.add('max-h-0', 'opacity-0');
            
            toggleLicenseRequirement(false); 
        }

        // --- Summary Page Generation ---

            const container = document.querySelector('.max-w-2xl');
            
            // Map form field names to user-friendly labels for the summary
            const labelMap = {
                full_name: "Full Name",
                email: "Email",
                country_code: "Country Code",
                phone_number: "Phone Number",
                living_eu_uk: "Residency in EU/UK",
                hold_license: "Valid Drone Pilot License",
                prior_experience_radio: "Prior Ag-Drone Experience",
                prior_experience: "Specified Drone Models",
                own_drone_radio: "Own Agricultural Drone",
                own_drone: "Owned Drone Model/Capacity",
                willing_to_travel: "Willingness to Travel (EU/UK)",
                self_sponsor_invest: "Visa Investment Willingness",
            };
            
            let summaryHTML = `
                <h1 class="text-3xl font-bold text-emerald-700 mb-2">âœ… Application Submitted Successfully!</h1>
                <p class="text-lg text-gray-600 mb-8">Thank you for applying. Your details have been received. We will contact you at ${data.email || 'the provided email address'}.</p>
                
                <h2 class="text-xl font-semibold text-gray-800 mb-4 border-b pb-2">Summary of Application Data:</h2>
                
                <div class="space-y-4">
            `;

            // Data contains the successful form submission details
            for (const [key, value] of Object.entries(data)) {
                // Skip internal form data and file data (as files aren't stored in this summary)
                if (key === 'action' || key.startsWith('attach_')) continue;
                
                // Skip conditional text fields if the answer was 'No'
                if (key === 'prior_experience' && data.prior_experience_radio === 'No') continue;
                if (key === 'own_drone' && data.own_drone_radio === 'No') continue;
                
                let displayLabel = labelMap[key] || key.replace(/_/g, ' ').toUpperCase();
                let displayValue = value.trim() || 'N/A';
                
                summaryHTML += `
                    <div class="flex border-b border-gray-100 pb-2">
                        <span class="font-medium text-gray-600 w-1/2">${displayLabel}:</span>
                        <span class="text-gray-900 w-1/2 font-bold">${displayValue}</span>
                    </div>
                `;
            }

            summaryHTML += `
                </div>
                <div class="pt-6 mt-6 border-t border-gray-200">
                    <button onclick="window.location.reload()" class="px-6 py-2 bg-emerald-600 text-white font-semibold rounded-full shadow-lg hover:bg-emerald-700 transition-colors duration-200">
                        Start a New Application
                    </button>
                </div>
            `;

            container.innerHTML = summaryHTML;
            container.classList.add('h-full');
        }


        // --- Submission Logic (Sends data to Flask backend on port 5000) ---

        async function submitForm() {
            // Final client-side validation check
            if (!validateStep(currentStep)) return; 

            document.getElementById('submission-error').classList.add('hidden');
            
            submitButton.disabled = true;
            submitButton.innerHTML = 'Submitting...';

            const formData = new FormData(form);
            const data = Object.fromEntries(formData.entries()); 

            try {
                // IMPORTANT: Ensure your Flask server is running on http://127.0.0.1:5000
                const response = await fetch('http://127.0.0.1:5000/submit', {
                    method: 'POST',
                    body: formData 
                });

                const result = await response.json();

                if (response.ok && result.success) {
                    console.log("Server success:", result.message);
                    generateSummaryPage(data);
                } else {
                    console.error("Server failure:", result.message);
                    
                    const emailField = document.getElementById('email');
                    
                    if (response.status === 409) { 
                        // Specific handling for unique email conflict from Flask
                        emailField.setCustomValidity(result.message);
                        emailField.reportValidity();
                    } else {
                        // General server errors
                        document.getElementById('submission-error').innerHTML = result.message;
                        document.getElementById('submission-error').classList.remove('hidden');
                    }
                }
            } catch (error) {
                console.error("Fetch error:", error);
                document.getElementById('submission-error').innerHTML = 'A network error occurred. Please ensure your Flask server is running on port 5000.';
                document.getElementById('submission-error').classList.remove('hidden');
            } finally {
                submitButton.disabled = false;
                submitButton.innerHTML = 'Submit Application';
            }
        }

        // --- Custom Input Filtering Handlers ---

        document.querySelectorAll('input').forEach(input => {
            const inputType = input.getAttribute('data-input-type');

            if (inputType === 'text-only') {
                input.addEventListener('input', (event) => {
                    event.target.value = event.target.value.replace(/[^a-zA-Z\s]/g, '');
                });
            } else if (inputType === 'number-only') {
                input.addEventListener('input', (event) => {
                    event.target.value = event.target.value.replace(/[^0-9]/g, '');
                });
            } else if (inputType === 'text-and-plus') {
                input.addEventListener('input', (event) => {
                    let value = event.target.value.trim();
                    if (value.startsWith('+')) {
                        event.target.value = '+' + value.substring(1).replace(/[^0-9]/g, '');
                    } else {
                        event.target.value = value.replace(/[^+0-9]/g, '').replace(/\+/g, '');
                    }
                });
            }
        });
